JDBC连接数据库示例

package cn.com.test.java.jdbc;

import lombok.extern.slf4j.Slf4j;

import java.sql.*;

/**
 * @Description
 * @Author Guang Ming Zhang
 * @Date 2021/5/26 9:07
 */
@Slf4j
public class JDBCTest {


    public static void main(String[] args) throws ClassNotFoundException, SQLException {
        String URL = "jdbc:mysql://cdb-q1uxszhm.bj.tencentcdb.com:10143/db_test_center?useSSL=false&useUnicode=true&characterEncoding=utf8&allowMultiQueries=true&serverTimezone=GMT%2B8";
        String USER = "root";
        String PASSWORD = "Gm859230";
        //1.加载驱动程序
        Class.forName("com.mysql.cj.jdbc.Driver");
        //2.获得数据库链接
        Connection conn = DriverManager.getConnection(URL, USER, PASSWORD);
        //3.通过数据库的连接操作数据库，实现增删改查（使用Statement类）
        Statement st = conn.createStatement();
        ResultSet rs = st.executeQuery("select * from user");
        //4.处理数据库的返回结果(使用ResultSet类)
        while (rs.next()) {
            System.out.println(rs.getString("user_name") + " "
                    + rs.getString("user_password"));
        }

        //关闭资源
        rs.close();
        st.close();
        conn.close();
    }


    //出参:
    //zhangsan 1`234567
    //lisi jdkjkejkfnmvk

}
